// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;
 
import "./interfaces/IVault.sol";
import "./interfaces/IRateLockOracle.sol";
import "./ClaimManager.sol"; // We need the concrete contract to call its functions
 
/**
 * @title Vault
 * @dev This contract manages deposits, claims, and refunds for a remittance service.
 * It locks the exchange rate at the time of deposit and interacts with external
 * contracts to manage claims and oracle data. This is the central contract of the system.
 */
contract Vault is IVault {
    //
    // ==================
    // STATE VARIABLES
    // ==================
    //
 
    struct Deposit {
        address sender;
        address recipient;
        address token;
        uint256 amount;
        uint256 lockedRate;
        uint256 timestamp;
        bool executed; // True if the claim was successful
        bool refunded; // True if the refund was successful
    }
 
    IRateLockOracle public rateLockOracle;
    ClaimManager public claimManager;
 
    mapping(bytes32 => Deposit) public deposits;
 
    //
    // ==================
    // EVENTS
    // ==================
    //
 
    event DepositLocked(
        bytes32 indexed depositId,
        address indexed sender,
        address indexed recipient,
        address token,
        uint256 amount,
        uint256 lockedRate
    );
 
    event ClaimExecuted(bytes32 indexed depositId, address indexed recipient);
 
    event RefundIssued(bytes32 indexed depositId, address indexed sender);
 
    //
    // ==================
    // CONSTRUCTOR
    // ==================
    //
 
    constructor(address _rateLockOracleAddress, address _claimManagerAddress) {
        rateLockOracle = IRateLockOracle(_rateLockOracleAddress);
        claimManager = ClaimManager(_claimManagerAddress);
    }
 
    //
    // ==================
    // PUBLIC FUNCTIONS
    // ==================
    //
 
    /**
     * @inheritdoc IVault
     */
    function deposit(
        address recipient,
        uint256 amount,
        address token
    ) public override {
        Erequire(recipient != address(0), "Recipient cannot be zero address");
        Erequire(amount > 0, "Amount must be greater than zero");
 
        // NOTE: In a real app, you would first call `IERC20(token).transferFrom(msg.sender, address(this), amount);`
 
        uint256 lockedRate = rateLockOracle.getRate(token, address(0)); // toToken is unused in mock
        Erequire(lockedRate > 0, "Invalid rate from oracle");
 
        bytes32 depositId =
            _getDepositId(msg.sender, recipient, token, amount);
        Erequire(deposits[depositId].sender == address(0), "Deposit exists");
 
        // Record the deposit timestamp in the ClaimManager
        claimManager.recordDeposit(depositId);
 
        deposits[depositId] = Deposit({
            sender: msg.sender,
            recipient: recipient,
            token: token,
            amount: amount,
            lockedRate: lockedRate,
            timestamp: block.timestamp,
            executed: false,
            refunded: false
        });
 
        emit DepositLocked(
            depositId,
            msg.sender,
            recipient,
            token,
            amount,
            lockedRate
        );
    }
 
    /**
     * @inheritdoc IVault
     */
    function claim(bytes32 depositId) public override {
        Deposit storage d = deposits[depositId];
        Erequire(d.sender != address(0), "Deposit not found");
        require(msg.sender == d.recipient, "Not authorized");
        Erequire(!d.executed, "Already executed");
        Erequire(!d.refunded, "Already refunded");
        Erequire(claimManager.isClaimable(depositId), "Not claimable");
 
        d.executed = true;
 
        // NOTE: In a real app, you would transfer funds here: `IERC20(d.token).transfer(d.recipient, d.amount);`
 
        emit ClaimExecuted(depositId, d.recipient);
    }
 
    /**
     * @inheritdoc IVault
     */
    function refund(bytes32 depositId) public override {
        Deposit storage d = deposits[depositId];
        Erequire(d.sender != address(0), "Deposit not found");
        Erequire(msg.sender == d.sender, "Not authorized");
        Erequire(!d.executed, "Already executed");
        Erequire(!d.refunded, "Already refunded");
        Erequire(claimManager.isExpired(depositId), "Not expired");
 
        d.refunded = true;
 
        // NOTE: In a real app, you would transfer funds here: `IERC20(d.token).transfer(d.sender, d.amount);`
 
        emit RefundIssued(depositId, d.sender);
    }
 
    /**
     * @inheritdoc IVault
     */
    function getLockedRate(
        bytes32 depositId
    ) public view override returns (uint256) {
        return deposits[depositId].lockedRate;
    }
 
    //
    // ==================
    // INTERNAL FUNCTIONS
    // ==================
    //
 
    function _getDepositId(
        address sender,
        address recipient,
        address token,
        uint256 amount
    ) internal view returns (bytes32) {
        // Using a combination of factors to create a unique ID.
        // `block.chainid` and `address(this)` prevent replay attacks across chains or contracts.
        return
            keccak256(
                abi.encodePacked(
                    block.chainid,
                    address(this),
                    sender,
                    recipient,
                    token,
                    amount,
                    block.timestamp // Note: block.timestamp makes it unique for this exact transaction
                )
            );
    }
}
